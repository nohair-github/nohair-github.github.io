Setting up a mailserver on OpenBSD 6.9: OpenSMTPD tweaks and testing

J. Further configuration of OpenSMTPD

The final smtpd.conf file is here. Changes from the initial file above are indicated in bold.

# smtpd.conf for nohair.online mailserver 8/15/21 gsb

# PKI keys for TLS
pki mail cert "/etc/ssl/primo.nohair.online.fullchain.pem"
pki mail key "/etc/ssl/private/primo.nohair.online.key"

# Macros
filters = " { check_rdns check_fcrdns rspamd } "

# Filters
filter check_rdns phase connect match !rdns disconnect "550 DNS error"
filter check_fcrdns phase connect match !fcrdns disconnect "550 DNS error"
filter "rspamd" proc-exec "/usr/local/libexec/smtpd/filter-rspamd"

# Tables
table aliases file:/etc/mail/aliases
table credentials passwd:/etc/mail/credentials
table virtuals file:/etc/mail/virtuals

# Listeners
listen on all tls pki "mail" hostname "mail.example.com" mask-src filter $filters
listen on egress port submission tls-require pki "mail" hostname "mail.example.com" \
  auth <credentials> filter "rspamd"

# Actions
action "local_mail" mbox alias <aliases>
action "domain_mail" maildir "/var/vmail/nohair.online/%{dest.user:lowercase}" virtual <virtuals>
action "outbound" relay

# Rules

# Local
match from local for local action "local_mail"
match from local for domain "nohair.online" action "domain_mail"
match from local for any action "outbound"

# Remote
match from any for domain "nohair.online" action "domain_mail"
match auth from any for any action "outbound"
